### 2.5 考勤机对接代码示例

```javascript
// Wiegand 26 协议解析
function parseWiegand26(data) {
  // Wiegand26格式: 1位校验 + 8位高位 + 16位低位 + 1位校验
  const facilityCode = (data >> 16) & 0xFF;  // 门禁编号
  const cardNumber = data & 0xFFFF;            // 卡号
  
  return {
    facilityCode,
    cardNumber,
    fullCardNo: `${facilityCode.toString().padStart(3, '0')}${cardNumber.toString().padStart(5, '0')}`
  };
}

// TCP/IP 考勤机数据接收 (Node.js)
const net = require('net');
const server = net.createServer((socket) => {
  socket.on('data', (buffer) => {
    const attendanceData = parseAttendancePacket(buffer);
    
    // 写入数据库
    saveAttendance({
      employee_no: attendanceData.cardNumber,
      check_time: attendanceData.timestamp,
      machine_no: attendanceData.deviceId,
      event_type: attendanceData.direction  // 0=进门, 1=出门
    });
  });
});

server.listen(5000, '0.0.0.0');
```

```javascript
// 分拣扫码流程
async function sortingScan(barcode) {
  // 1. 解析二维码
  const orderData = parseQRCode(barcode);
  
  // 2. 查询订单信息
  const order = await getOrderWithPackage(orderData.order_no);
  
  // 3. 执行分拣规则
  const sortResult = executeSortingRules(order);
  
  // 4. 组装播报内容
  const announcement = {
    area: sortResult.sort_area,
    position: sortResult.sort_position,
    order_no: order.order_no,
    customer: order.customer_name,
    package_count: order.package_count,
    message: `订单${order.order_no}，客户${order.customer_name}，请放置${sortResult.sort_area}区${sortResult.sort_position}`
  };
  
  // 5. 打印标签
  printSortLabel(announcement);
  
  // 6. 语音播报
  playAnnouncement(announcement.message);
  
  // 7. 发送LED显示指令
  sendLEDDisplay(announcement.area, announcement.position);
  
  return announcement;
}

// 语音播报函数
function playAnnouncement(message) {
  // 方式1: 调用Windows TTS
  const { exec } = require('child_process');
  exec(`powershell -command "Add-Type -AssemblyName System.Speech; $speak = New-Object System.Speech.Synthesis.SpeechSynthesizer; $speak.Speak('${message}')"`);
  
  // 方式2: 发送串口指令给工业扬声器
  serialPort.write(`AT+PLAY=${message}\r\n`);
  
  // 方式3: 调用网络语音API
  fetch('http://speaker.local/play', {
    method: 'POST',
    body: JSON.stringify({ message })
  });
}
```
