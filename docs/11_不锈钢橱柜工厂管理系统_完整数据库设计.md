## 三、数据表设计（完整版）

### 3.1 经销商模块

#### 3.1.1 经销商主表 (dealer)

```sql
-- ============================================
-- 经销商主表：存储经销商基本信息
-- ============================================
CREATE TABLE dealer (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    dealer_code VARCHAR(50) NOT NULL UNIQUE COMMENT '经销商编号：JXS+序号',
    dealer_name VARCHAR(200) NOT NULL COMMENT '经销商名称',
    dealer_type VARCHAR(20) NOT NULL DEFAULT 'authorized' 
        COMMENT '类型：direct/authorized/retail/experience',
    
    -- 基本信息
    legal_person VARCHAR(100) COMMENT '法人代表',
    contact_person VARCHAR(100) COMMENT '联系人',
    phone VARCHAR(20) NOT NULL,
    mobile VARCHAR(20),
    email VARCHAR(100),
    
    -- 地址
    province VARCHAR(50),
    city VARCHAR(50),
    district VARCHAR(50),
    address VARCHAR(500),
    
    -- 资质
    business_license VARCHAR(500) COMMENT '营业执照',
    tax_no VARCHAR(50) COMMENT '税号',
    bank_name VARCHAR(100) COMMENT '开户银行',
    bank_account VARCHAR(50) COMMENT '银行账号',
    
    -- 合同
    contract_no VARCHAR(50),
    contract_start DATE,
    contract_end DATE,
    commission_rate DECIMAL(5,2) COMMENT '佣金比例%',
    
    -- 状态
    status VARCHAR(20) DEFAULT 'pending',
    activation_date DATE,
    
    -- 额度
    credit_limit DECIMAL(12,2) DEFAULT 0,
    credit_used DECIMAL(12,2) DEFAULT 0,
    
    -- 组织
    parent_dealer_id UUID REFERENCES dealer(id),
    
    -- 系统账号
    admin_user_id UUID,
    
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_dealer_status ON dealer(status);
CREATE INDEX idx_dealer_type ON dealer(dealer_type);
```
